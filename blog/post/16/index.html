#!/bin/sh
# 转发到 alacritty，支持 --working-directory 参数
while [ $# -gt 0 ]; do
    case "$1" in
        --working-directory)
            shift
            WD="$1"
            shift
            ;;
        *)
            shift
            ;;
    esac
done
exec alacritty ${WD:+--working-directory "$WD"}
